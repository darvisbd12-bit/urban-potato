function showToast(msg){

  t.classList.add('show');
  clearTimeout(showToast._tm);
  showToast._tm = setTimeout(()=>t.classList.remove('show'), 2400);

/* ---------- Render catálogo ---------- */
function getFilteredProducts(){
  let list = products.slice();
  if(currentCat !== 'todas') list = list.filter(p=>p.cat===currentCat);
  if(searchTerm.trim()) list = list.filter(p=>p.name.toLowerCase().includes(searchTerm.toLowerCase()));
  if(sortMode==='price-asc') list.sort((a,b)=>a.price-b.price);
  if(sortMode==='price-desc') list.sort((a,b)=>b.price-a.price);
  if(sortMode==='name-asc') list.sort((a,b)=>a.name.localeCompare(b.name));
  return list;
}

function renderCatalog(){
  const grid = document.getElementById('productGrid');
  const list = getFilteredProducts();
  document.getElementById('resultsCount').textContent = list.length + (list.length===1?' bicicleta':' bicicletas');

  if(list.length===0){
    grid.innerHTML = '<div class="empty-msg" style="grid-column:1/-1;">No encontramos bicicletas con esos filtros.</div>';
    return;
  }

  grid.innerHTML = list.map(p=>{
    const color = CAT_COLORS[p.cat];
    const noStock = p.stock<=0;
    return `
    <div class="card">
      <div class="punch"></div>
      <div class="card-media" style="background:${color}1A;">${bikeIcon(color)}</div>
      <div class="dash-line"></div>
      <div class="card-body">
        <div class="tag-cat" style="color:${color};">${CAT_LABELS[p.cat]}</div>
        <h3>${p.name}</h3>
        <div class="card-desc">${p.desc}</div>
        <div class="card-foot">
          <span class="price mono">${money(p.price)}</span>
          <span class="stock ${p.stock<=2 && p.stock>0?'low':''}">${noStock?'agotado':p.stock+' disp.'}</span>
        </div>
        <button class="add-btn" ${noStock?'disabled':''} onclick="addToCart('${p.id}')">${noStock?'Sin existencias':'Agregar al carrito'}</button>
      </div>
    </div>`;
  }).join('');
}

/* ---------- Carrito ---------- */
function addToCart(id){
  const p = products.find(x=>x.id===id);
  if(!p || p.stock<=0) return;
  const inCart = cart[id]||0;
  if(inCart >= p.stock){ showToast('No hay más existencias disponibles'); return; }
  cart[id] = inCart+1;
  renderCart();
  showToast(p.name + ' agregada al carrito');
}
function changeQty(id, delta){
  const p = products.find(x=>x.id===id);
  let qty = (cart[id]||0) + delta;
  if(qty<=0){ delete cart[id]; }
  else if(qty > p.stock){ showToast('No hay más existencias disponibles'); return; }
  else { cart[id] = qty; }
  renderCart();
}
function removeFromCart(id){
  delete cart[id];
  renderCart();
}
function cartTotalItems(){
  return Object.values(cart).reduce((a,b)=>a+b,0);
}
function renderCart(){
  document.getElementById('cartCount').textContent = cartTotalItems();
  const wrap = document.getElementById('cartItems');
  const ids = Object.keys(cart);

  if(ids.length===0){
    wrap.innerHTML = '<div class="empty-msg">Tu carrito está vacío.</div>';
  } else {
    wrap.innerHTML = ids.map(id=>{
      const p = products.find(x=>x.id===id);
      if(!p) return '';
      const color = CAT_COLORS[p.cat];
      return `
      <div class="cart-item">
        <div class="ci-media" style="background:${color}1A;">${bikeIcon(color)}</div>
        <div class="ci-info">
          <h4>${p.name}</h4>
          <div class="ci-price mono">${money(p.price)}</div>
          <div class="qty-stepper">
            <button onclick="changeQty('${id}',-1)">−</button>
            <span>${cart[id]}</span>
            <button onclick="changeQty('${id}',1)">+</button>
          </div>
        </div>
        <button class="remove-x" onclick="removeFromCart('${id}')">Quitar</button>
      </div>`;
    }).join('');
  }

  const subtotal = ids.reduce((sum,id)=>{
    const p = products.find(x=>x.id===id);
    return sum + (p ? p.price*cart[id] : 0);
  },0);
  const shipping = subtotal>0 ? (subtotal>4000?0:60) : 0;
  const total = subtotal + shipping;

  document.getElementById('subtotalVal').textContent = money(subtotal);
  document.getElementById('shippingVal').textContent = shipping===0 ? 'Gratis' : money(shipping);
  document.getElementById('totalVal').textContent = money(total);
  document.getElementById('checkoutBtn').disabled = ids.length===0;
}

function startEdit(id){
  const p = products.find(x=>x.id===id);
  if(!p) return;
  editingId = id;
  document.getElementById('f-name').value = p.name;
  document.getElementById('f-cat').value = p.cat;
  document.getElementById('f-price').value = p.price;
  document.getElementById('f-stock').value = p.stock;
  document.getElementById('f-desc').value = p.desc;
  document.getElementById('formTitle').textContent = 'Editar bicicleta';
  document.getElementById('saveProductBtn').textContent = 'Guardar cambios';
  document.getElementById('cancelEditBtn').style.display = 'inline-block';
  window.scrollTo({top:0,behavior:'smooth'});
}

function resetForm(){
  editingId = null;
  ['f-name','f-price','f-stock','f-desc'].forEach(id=>document.getElementById(id).value='');
  document.getElementById('f-cat').value='montana';
  document.getElementById('formTitle').textContent = 'Agregar bicicleta';
  document.getElementById('saveProductBtn').textContent = 'Guardar';
  document.getElementById('cancelEditBtn').style.display = 'none';
}
document.getElementById('cancelEditBtn').onclick = resetForm;

document.getElementById('saveProductBtn').onclick = ()=>{
  const name = document.getElementById('f-name').value.trim();
  const cat = document.getElementById('f-cat').value;
  const price = Number(document.getElementById('f-price').value);
  const stock = Number(document.getElementById('f-stock').value);
  const desc = document.getElementById('f-desc').value.trim();

  if(!name || !price || price<=0 || stock<0 || isNaN(stock)){
    showToast('Completa nombre, precio y existencias válidas');
    return;
  }

  if(editingId){
    const p = products.find(x=>x.id===editingId);
    Object.assign(p, {name, cat, price, stock, desc});
    showToast('Bicicleta actualizada');
  } else {
    const id = 'b' + (Date.now());
    products.push({id, name, cat, price, stock, desc});
    showToast('Bicicleta agregada al inventario');
  }
  resetForm();
  renderAdminTable();
  renderCatalog();
};

function deleteProduct(id){
  if(!confirm('¿Eliminar esta bicicleta del inventario?')) return;
  products = products.filter(p=>p.id!==id);
  delete cart[id];
  renderAdminTable();
  renderCatalog();
  renderCart();
  showToast('Bicicleta eliminada');
}
